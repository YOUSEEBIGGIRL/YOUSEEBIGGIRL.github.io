LeetCode 43. 字符串相乘

2021-06-23
4分钟阅读时长

给定两个以字符串形式表示的非负整数 num1 和 num2，返回 num1 和 num2 的乘积，它们的乘积也表示为字符串形式。

给定两个以字符串形式表示的非负整数 num1 和 num2，返回 num1 和 num2 的乘积，它们的乘积也表示为字符串形式。

示例 1:

输入: num1 = “2”, num2 = “3” 输出: “6”

示例 2:

输入: num1 = “123”, num2 = “456” 输出: “56088”

说明：

num1 和 num2 的长度小于110。 num1 和 num2 只包含数字 0-9。 num1 和 num2 均不以零开头，除非是数字 0 本身。 不能使用任何标准库的大数类型（比如 BigInteger）或直接将输入转换为整数来处理。

方法1 模拟数学

在草稿纸上写一下乘法的运算步骤，并将其改写为代码形式：

// 		 45
//	*  123
// --------------	
//     135							
//  +  90        
//    45
// ---------------
//    5535
func multiply(num1 string, num2 string) string {
    var (
		flag int             // 保存进位
		res  string 
		pown int             // 确定要补几个 0
	)
    
    if num1 == "0" || num2 == "0" {
        return "0"
    }

	for i := len(num1) - 1; i >= 0; i-- {
		var r strings.Builder
        
		for j := len(num2) - 1; j >= 0; j-- {
			v1 := int(num1[i] - '0')
			v2 := int(num2[j] - '0')

			tmp := v1*v2 + flag
			flag = tmp / 10
			tmp %= 10

			r.WriteByte(byte(tmp + '0'))
		}

		if flag != 0 {
			r.WriteByte(byte(flag) + '0')
			flag = 0
		}

		rstr := reverseStr(r.String())

		// 		 45
		//	*  123
		// --------------	每步后面补 n-1 个 0，n 为 位数
		//     135							135	 补 0 个
		//     90         =>  	900	 补 1 个
		//    45			   			 4500  补 2 个
		for i := 0; i < pown; i++ {
			rstr += "0"
		}
		pown++

    // 字符串相加
		res = strAdd(res, rstr)
	}
    
	return res
}

func reverseStr(s string) string {
	b := []byte(s)
	for i, j := 0, len(s)-1; i < j; i, j = i+1, j-1 {
		b[i], b[j] = b[j], b[i]
	}
	return string(b)
}

func strAdd(s1, s2 string) string {
	var (
		flag int
		res strings.Builder
	)
	for i, j := len(s1)-1, len(s2)-1; i >= 0 || j >= 0; i, j = i-1, j-1 {
		var v1, v2 int
		if i >= 0 {
			v1 = int(s1[i] - '0')
		} else {
			v1 = 0
		}

		if j >= 0 {
			v2 = int(s2[j] - '0')
		} else {
			v2 = 0
		}

		sum := v1 + v2 + flag
		flag = sum / 10
		sum %= 10

		res.WriteByte(byte(sum) + '0')
	}

	if flag != 0 {
		res.WriteByte(byte(flag) + '0')
	}

	str := reverseStr(res.String())
	return str
}

方法2 优化

方法一的做法是从右往左遍历乘数，将乘数的每一位与被乘数相乘得到对应的结果，再将每次得到的结果累加，整个过程中涉及到较多字符串相加的操作。如果使用数组代替字符串存储结果，则可以减少对字符串的操作。具体的方法如下：

代码如下：

go

func multiply(num1 string, num2 string) string {
    if num1 == "0" || num2 == "0" {
        return "0"
    }
    
    n1l := len(num1)
    n2l := len(num2)
    
    n := make([]int, n1l+n2l, n1l+n2l)
    
    for i := n1l - 1; i >= 0; i-- {
        v1 := num1[i] - '0'
        for j := n2l - 1; j >= 0; j-- {
            v2 := num2[j] - '0'
            n[i+j+1] += int(v1) * int(v2)
            
        }
    }
    
    var (
        flag  int
        res   strings.Builder
        start int
    )    
        
    for i := len(n) - 1; i >= 0; i-- {
        n[i] += flag
        flag = n[i] / 10
        n[i] %= 10
    }
    
    //fmt.Println(n)
    
    if n[0] == 0 {
        start = 1
    }
    
    for ; start < len(n); start++ {
        s := strconv.Itoa(n[start])
        res.WriteString(s)
    }
    
    return res.String()
}

cpp

class Solution {
public:
    string multiply(string num1, string num2) {
        if (num1 == "0" || num2 == "0") {
            return "0";
        }
        int n1l = num1.size();
        int n2l = num2.size();
        vector<int> v(n1l + n2l);
        
        for (int i = n1l - 1; i >= 0; i--) {
            for (int j = n2l - 1; j >= 0; j--) {
                v[i+j+1] += (num1[i] - '0') * (num2[j] - '0');
            }
        }

        int flag = 0;
        string res;
        int start = 0;
        
        for (int i = v.size() - 1; i >= 0; i--) {
            v[i] += flag;
            flag = v[i] / 10;
            v[i] %= 10;
        }
        
        // for (auto vv : v) {
        //     cout << vv << endl;
        // }
        
        if (v[0] == 0) {
            start = 1;
        }

        
        for (; start < v.size(); start++) {
            res += char(v[start]+'0');    
        }
        
        return res;
    }
};
